Quick start with docker
# 1. Copy environment file
cp docker-compose.env.example .env

# 2. Edit .env and update:
#    - POSTGRES_PASSWORD (set a strong password)
#    - SECRET_KEY (set a secure JWT key)
#    - NEXT_PUBLIC_API_URL (your backend URL)

# 3. Start everything
docker compose up -d
or
sudo docker compose up -d

Useful commands
# Stop all services
sudo docker compose stop

# Start services
sudo docker compose start

# Stop and remove containers
sudo docker compose down

# Stop, remove containers, and delete volumes (⚠️ deletes database)
sudo docker compose down -v

# Rebuild after code changes
sudo docker compose build
sudo docker compose up -d
sudo docker compose up -d --build frontend

# View logs
sudo docker compose logs -f

# Stop services
sudo docker compose stop

# Restart services
sudo docker compose restart

# View specific service logs
sudo docker compose logs -f backend
sudo docker compose logs -f frontend

# Or use the quick start script
./docker-start.sh

# Stop and remove all existing containers
sudo docker compose down

# Or if that doesn't work, remove them individually:
sudo docker stop synvoy-db synvoy-backend synvoy-frontend synvoy-nginx
sudo docker rm synvoy-db synvoy-backend synvoy-frontend synvoy-nginx
